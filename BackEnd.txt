/server
  /auth                       # Authentication-related logic (OAuth)
    - authController.js        # Handles OAuth callback and user info
    - authRoutes.js            # Routes for OAuth login and callback
    - authService.js           # OAuth integration logic (token exchange, validation)
  /api                        # API logic (garage data, user data, etc.)
    /controllers              # Logic to handle API requests
      - garageController.js    # Controller for handling garage API routes
      - userController.js      # Controller for user-related API logic
    /models                   # Database models (e.g., Garage, User)
      - Garage.js              # Mongoose schema for garage
      - User.js                # Mongoose schema for user data (OAuth data)
    /routes                   # API route definitions
      - garageRoutes.js        # Routes related to garages (GET, POST, etc.)
      - userRoutes.js          # Routes related to users (GET, POST, etc.)
    /services                  # Business logic and utility functions
      - garageService.js       # Logic for garage data (fetch, add, update)
      - userService.js         # Logic for handling user-related operations (OAuth)
  server.js                   # Entry point for backend app (Express setup)
  .env                        # Environment variables for backend (API keys, OAuth secrets)
  package.json                # Backend dependencies and scripts
